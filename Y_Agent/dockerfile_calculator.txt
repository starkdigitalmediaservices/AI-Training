# Unit Converter Agent Dockerfile
FROM python:3.9-slim

# Environment for unbuffered logs and no pip cache
ENV PYTHONDONTWRITEBYTECODE=1 \
    PYTHONUNBUFFERED=1 \
    PIP_NO_CACHE_DIR=1

# Set working directory
WORKDIR /app

# Install system dependencies
RUN apt-get update && apt-get install -y \
    curl \
    && rm -rf /var/lib/apt/lists/*

# Install Python dependencies
RUN pip install --upgrade pip && pip install --no-cache-dir flask flask-cors requests python-dotenv

# Copy application code
COPY unit_converter_network.py .
COPY templates/ templates/
COPY static/ static/

# Expose port
EXPOSE 5002

# Add health check
HEALTHCHECK --interval=30s --timeout=30s --start-period=5s --retries=3 \
    CMD curl -f http://localhost:5002/health || exit 1

# Run the application
CMD ["python", "-u", "unit_converter_network.py"]
